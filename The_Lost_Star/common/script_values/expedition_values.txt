expedition_success_chance_value = {
	activity_host = {
		### HOST
		# SKILLS
		add = {
			value = learning
			divide = 2
			ceiling = yes
			max = 15
			desc = learning_modifier
		}
		add = {
			value = martial
			divide = 12
			ceiling = yes
			max = 10
			desc = martial_modifier
		}
		add = {
			value = prowess
			divide = 6
			ceiling = yes
			max = 15
			desc = prowess_modifier
		}
		# TRAIT

		# MODIFIERS

		# Antiquarian
		if = {
			limit = { exists = court_position:antiquarian_court_position }
			add = {
				value = 0
				court_position:antiquarian_court_position = {
					if = {
						limit = {
							aptitude = { court_position = antiquarian_court_position value <= 1 }
						}
						add = 3
					}
					else_if = {
						limit = {
							aptitude = { court_position = antiquarian_court_position value = 2 }
						}
						add = 6
					}
					else_if = {
						limit = {
							aptitude = { court_position = antiquarian_court_position value = 3 }
						}
						add = 9
					}
					else_if = {
						limit = {
							aptitude = { court_position = antiquarian_court_position value = 4 }
						}
						add = 12
					}				
					else = { add = 15 }
				}
				desc = expedition_success_chance.antiquarian
			}
		}

		# ESTATE
	}
	# GUESTS

	# OPTION

	# METHOD	

	# LOCATION

	# CULTURE

	# SIGHTINGS

	# MODIFIERS

	# EVENTS
	if = {
		limit = { exists = var:expedition_success_chance_event }
		add = {
			value = var:expedition_success_chance_event
			desc = expedition_success_chance.events
		}
	}
	# TYPE

}
